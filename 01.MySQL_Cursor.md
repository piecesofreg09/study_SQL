```sql
-- CREATE TABLE tt(ID int);
DELIMITER //
CREATE PROCEDURE cur()
BEGIN
  DECLARE done INT DEFAULT FALSE;
  DECLARE a CHAR(16);
  DECLARE b, c INT;
  DECLARE cur1 CURSOR 
  FOR SELECT PersonID FROM persons;
  DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = TRUE;
  
  OPEN cur1;
  
  -- LOOP NAME
  looppp: LOOP
  fetch cur1 into b;
  
  IF done THEN
	leave looppp;
  END IF;
  
  SELECT concat('myvar is ', b);
  IF b > 1 Then
	insert INTO tt values (b);
  END IF;
  END LOOP;
  
  close cur1;
END//
CALL cur();
SELECT * FROM tt;
```
